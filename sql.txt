-- Create database
CREATE DATABASE IF NOT EXISTS techtrail2025;
USE techtrail2025;

-- Create registrations table
CREATE TABLE IF NOT EXISTS registrations (
    id INT AUTO_INCREMENT PRIMARY KEY,
    reg_id VARCHAR(20),
    fullname VARCHAR(100),
    email VARCHAR(100),
    phone VARCHAR(20),
    college VARCHAR(100),
    department VARCHAR(50),
    year_of_study VARCHAR(20),
    topic VARCHAR(50),
    mode VARCHAR(20),
    comments TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Fix for older rows that do not have reg_id (optional)
UPDATE registrations
SET reg_id = CONCAT('TT2025-', LPAD(id, 3, '0'))
WHERE reg_id IS NULL OR reg_id = '';
